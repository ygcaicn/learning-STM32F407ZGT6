	 //	PE2-----------KEY2
	 //	PE3-----------KEY1
	 //	PE4-----------KEY0
	 //	PA0-----------WK_UP
	 //说明：
	 //		KEY0,KEY1,KEY2低电平有效（按下）
	 //		WK_UP 高电平有效（按下）

	 //
